6 Criando o Slice (Actions + Reducer)

O Slice √© onde a m√°gica acontece! Ele combina actions e reducer em um √∫nico lugar:

import type { Item, ItemsState, CreateItemPayload, UpdateItemPayload } from './types';

// ===========================================
// ESTADO INICIAL
// ===========================================
const initialState: ItemsState = {
  items: [],
  loading: false,
  error: null,
};

// ===========================================
// HELPER: Gerar ID √∫nico
// ===========================================
const generateId = (): string => {
  return Date.now().toString(36) + Math.random().toString(36).substring(2);
};

// ===========================================
// CRIA√á√ÉO DO SLICE
// ===========================================
const itemsSlice = createSlice({
  // Nome do slice (usado no devtools e para gerar action types)
  name: 'items',
  
  // Estado inicial
  initialState,
  
  // Reducers: fun√ß√µes que modificam o estado
  reducers: {
    
    // ==================
    // 1. ADICIONAR ITEM
    // ==================
    addItem: (state, action: PayloadAction<CreateItemPayload>) => {
      // Criar novo item com ID e data gerados automaticamente
      const newItem: Item = {
        id: generateId(),
        name: action.payload.name,
        description: action.payload.description,
        completed: false,
        createdAt: new Date().toISOString(),
      };
      
      // Push direto! Immer cuida da imutabilidade por baixo dos panos
      state.items.push(newItem);
    },

    // ==================
    // 2. REMOVER ITEM
    // ==================
    removeItem: (state, action: PayloadAction<string>) => {
      // action.payload √© o ID do item a remover
      state.items = state.items.filter(
        (item) => item.id !== action.payload
      );
    },

    // ==================
    // 3. ATUALIZAR ITEM
    // ==================
    updateItem: (state, action: PayloadAction<UpdateItemPayload>) => {
      // Encontrar o √≠ndice do item a atualizar
      const index = state.items.findIndex(
        (item) => item.id === action.payload.id
      );
      
      // Se encontrou, aplicar as mudan√ßas
      if (index !== -1) {
        state.items[index] = {
          ...state.items[index],
          ...action.payload.changes,
        };
      }
    },

    // ==================
    // 4. TOGGLE COMPLETED
    // ==================
    toggleCompleted: (state, action: PayloadAction<string>) => {
      const item = state.items.find(
        (item) => item.id === action.payload
      );
      
      if (item) {
        item.completed = !item.completed;
      }
    },

    // ==================
    // 5. LIMPAR TODOS
    // ==================
    clearAllItems: (state) => {
      state.items = [];
    },

    // ==================
    // 6. SET ERROR
    // ==================
    setError: (state, action: PayloadAction<string | null>) => {
      state.error = action.payload;
    },
  },
});

// ===========================================
// EXPORTA√á√ïES
// ===========================================

// Exportar as actions (para usar com dispatch)
export const { 
  addItem, 
  removeItem, 
  updateItem, 
  toggleCompleted, 
  clearAllItems,
  setError
} = itemsSlice.actions;

// Exportar o reducer (para usar na store)
export default itemsSlice.reducer;
